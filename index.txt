# 创建用户表user
CREATE TABLE IF NOT EXISTS user (
    id SMALLINT UNSIGNED,
    name VARCHAR(20),
    age TINYINT UNSIGNED,
    sex ENUM('男','女','保密'),
    tel INT UNSIGNED,
    email VARCHAR(50),
    address VARCHAR(200),
    birth YEAR,
    salary FLOAT(8,6),
    married TINYINT(1) COMMENT '0: 未结婚, 非0: 已结婚'
)ENGINE=InnoDB CHARSET=UTF8;
